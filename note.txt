const express = require('express');
const multer = require('multer');
const path = require('path');
const cors = require('cors'); // To handle cross-origin requests from React

const app = express();
const port = 3000;

// Enable CORS for all routes
app.use(cors());

// Configure multer storage
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    // Specify the directory where files will be stored
    cb(null, 'uploads/');
  },
  filename: (req, file, cb) => {
    // Save the file with a timestamp to avoid name collisions
    cb(null, `${Date.now()}-${file.originalname}`);
  }
});

// Initialize multer with the storage configuration
const upload = multer({ storage: storage });

// Route to handle file uploads
app.post('/upload', upload.single('file'), (req, res) => {
  if (!req.file) {
    return res.status(400).json({ message: 'No file uploaded' });
  }

  // Respond with success and file details
  res.status(200).json({
    message: 'File uploaded successfully',
    file: {
      filename: req.file.filename,
      path: req.file.path,
      size: req.file.size,
      mimeType: req.file.mimetype
    }
  });
});

// Start the server
app.listen(port, () => {
  console.log(`Server is running at http://localhost:${port}`);
});
